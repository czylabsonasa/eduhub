% hatvany
function [x,l,k]=hatvany(A,x,e)
  x=x/norm(x);
  l0=dot(A*x,x);
  l=l0;
  for k=1:1000
    x=A*x;
    x=x/norm(x);
    l1=l;
    l=dot(A*x,x);
    if abs(l1-l)<=e*(1+l0), break; end
  end
  if norm(A*x-l*x)^2>e, k=-k; end
end

% inverz iteracio (PLU nelkul)
function [x,l,k]=invit(A,x,e)
  sol=@(x) A\x;
  
  x=x/norm(x);
  l0=dot(sol(x),x);
  l=l0;
  for k=1:1000
    x=sol(x);
    x=x/norm(x);
    l1=l;
    l=dot(sol(x),x);
    if abs(l1-l)<=e*(1+l0), break; end
  end
  if norm(sol(x)-l*x)^2>e, k=-k; end
end
